      def define_paperclip_callbacks(*callbacks)
        define_callbacks *[callbacks, {:terminator => "result == false"}].flatten
        callbacks.each do |callback|
          eval "def before_\#{callback}(*args, &blk)\nset_callback(:\#{callback}, :before, *args, &blk)\nend\ndef after_\#{callback}(*args, &blk)\nset_callback(:\#{callback}, :after, *args, &blk)\nend\n"
        end
      end
