      def expiring_url(time = 3600, style_name = default_style)
        if path
          base_options = { :expires => time, :secure => use_secure_protocol?(style_name) }
          s3_object(style_name).url_for(:read, base_options.merge(s3_url_options)).to_s
        end
      end
